// AI生成：用户登录接口测试套件
describe('POST /api/auth/login', () => {
  // 正常路径：有效凭据登录成功
  test('should return JWT token with valid credentials', async () => {
    const res = await request(app)
      .post('/api/auth/login')
      .send({ username: 'testuser', password: 'Test@123' });
    expect(res.status).toBe(200);
    expect(res.body).toHaveProperty('token');
    expect(res.body.token).toMatch(/^eyJ/);
  });

  // 边界值：密码长度恰好为最小值8位
  test('should accept password with minimum length (8 chars)', async () => {
    const res = await request(app)
      .post('/api/auth/login')
      .send({ username: 'testuser', password: 'Test@123' });
    expect(res.status).toBe(200);
  });

  // 异常路径：错误密码返回401
  test('should return 401 with wrong password', async () => {
    const res = await request(app)
      .post('/api/auth/login')
      .send({ username: 'testuser', password: 'wrongpass' });
    expect(res.status).toBe(401);
    expect(res.body.message).toBe('Invalid credentials');
  });

  // 边界值：空用户名字段
  test('should return 400 when username is empty', async () => {
    const res = await request(app)
      .post('/api/auth/login')
      .send({ username: '', password: 'Test@123' });
    expect(res.status).toBe(400);
  });

  // 异常路径：SQL注入攻击防护
  test('should reject SQL injection in username', async () => {
    const res = await request(app)
      .post('/api/auth/login')
      .send({ username: "admin' OR '1'='1", password: 'any' });
    expect(res.status).toBe(401);
  });
});

import pytest
from cart import Cart  # 此时 Cart 类还不存在

class TestCartDiscount:
    def setup_method(self):
        self.cart = Cart()

    def test_no_discount_below_100(self):
        """总价不足100元，无折扣"""
        self.cart.add_item('book', 50, 1)
        assert self.cart.total() == 50.0

    def test_10_percent_discount_over_100(self):
        """总价100-500元，享9折"""
        self.cart.add_item('phone', 200, 1)
        assert self.cart.total() == 180.0  # 200 * 0.9

    def test_20_percent_discount_over_500(self):
        """总价500元以上，享8折"""
        self.cart.add_item('laptop', 600, 1)
        assert self.cart.total() == 480.0  # 600 * 0.8

    def test_empty_cart_returns_zero(self):
        """空购物车总价为0"""
        assert self.cart.total() == 0.0

# Step 2: AI 根据测试生成最简实现（绿灯阶段）
class Cart:
    def __init__(self):
        self.items = []

    def add_item(self, name: str, price: float, qty: int):
        self.items.append({'name': name, 'price': price, 'qty': qty})

    def total(self) -> float:
        raw = sum(i['price'] * i['qty'] for i in self.items)
        if raw >= 500: return raw * 0.8
        if raw >= 100: return raw * 0.9
        return raw

// AI生成：商品查询接口 - 数据驱动测试
const testCases = [
  // 正常场景
  { id: 1,       expectedStatus: 200, expectedName: '苹果手机' },
  // 边界值：最大有效ID
  { id: 999999,  expectedStatus: 200, expectedName: '边界商品' },
  // 异常：商品不存在
  { id: 0,       expectedStatus: 404, expectedError: 'Product not found' },
  // 异常：负数ID
  { id: -1,      expectedStatus: 400, expectedError: 'Invalid product id' },
  // 安全：超大数字
  { id: 9999999999, expectedStatus: 400, expectedError: 'Id out of range' },
];

testCases.forEach(({ id, expectedStatus, expectedName, expectedError }) => {
  pm.test(`GET /products/${id} → ${expectedStatus}`, () => {
    const res = pm.response;
    pm.expect(res.code).to.equal(expectedStatus);

    if (expectedStatus === 200) {
      const body = res.json();
      pm.expect(body.name).to.equal(expectedName);
      pm.expect(body).to.have.property('price').that.is.a('number');
      pm.expect(body).to.have.property('stock').that.is.at.least(0);
      // 验证响应时间 SLA
      pm.expect(res.responseTime).to.be.below(500);
    } else {
      pm.expect(res.json().message).to.include(expectedError);
    }
  });
});

// Page Object: 商品详情页
class ProductPage {
  constructor(page) {
    this.page = page;
    this.addToCartBtn = page.getByRole('button', { name: '加入购物车' });
    this.cartCount    = page.locator('[data-testid="cart-count"]');
    this.price        = page.locator('[data-testid="product-price"]');
  }
  async addToCart() {
    await this.addToCartBtn.click();
    // 等待购物车数量更新
    await this.page.waitForFunction(() =>
      document.querySelector('[data-testid="cart-count"]')?.textContent !== '0'
    );
  }
}

// Page Object: 结算页
class CheckoutPage {
  constructor(page) {
    this.page = page;
    this.orderTotal = page.locator('[data-testid="order-total"]');
    this.payBtn     = page.getByRole('button', { name: '立即支付' });
  }
  async getTotal() {
    return parseFloat((await this.orderTotal.textContent()).replace('¥', ''));
  }
}

// AI 生成的 E2E 测试：完整购物流程
test('用户完成商品购买全流程', async ({ page }) => {
  const productPage  = new ProductPage(page);
  const checkoutPage = new CheckoutPage(page);

  // 1. 访问商品详情页
  await page.goto('https://shop.example.com/products/1');
  const price = parseFloat((await productPage.price.textContent()).replace('¥', ''));

  // 2. 加入购物车
  await productPage.addToCart();
  await expect(productPage.cartCount).toHaveText('1');

  // 3. 进入结算页
  await page.goto('https://shop.example.com/checkout');
  const total = await checkoutPage.getTotal();
  expect(total).toBeCloseTo(price * 0.9, 1); // 验证9折优惠

  // 4. 提交订单（Mock 支付）
  await checkoutPage.payBtn.click();
  await expect(page).toHaveURL(/\/order\/success/);
  await expect(page.locator('h1')).toContainText('订单已提交');
});

💬 思考与讨论
一个 E2E 测试在流水线中随机失败（Flaky Test），作为测试工程师智能体你会怎么处理？

处理 Flaky Test 的 AI 策略：①分析历史失败日志，识别失败是否与特定时间/环境/并发有关；②检查测试代码，定位可能的异步等待问题（硬编码 sleep 而非智能等待）；③建议修复方案：替换为 Playwright 的 waitForSelector/waitForResponse；④对于无法立即修复的抖动测试，先打上 @flaky 标签隔离，避免干扰 CI 门禁，同时创建缺陷单追踪。

测试报告与缺陷管理

测试结果的价值在于快速传达质量信号。AI 测试智能体不仅能生成直观的测试报告，还能对失败用例进行智能分析，将原始堆栈信息转化为「人类可读的缺陷描述」，大幅减少缺陷分析时间。

1
